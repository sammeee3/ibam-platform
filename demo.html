    <script>

        // Page Navigation

        function showPage(pageId) {

            const pages = document.querySelectorAll('.page');

            pages.forEach(page => page.classList.remove('active'));

            

            const targetPage = document.getElementById(pageId);

            if (targetPage) {

                targetPage.classList.add('active');

                window.scrollTo(0, 0);

            }

        }


        // Assessment Routing

        function routeToPath(path) {

            // Add visual feedback

            event.target.style.background = 'var(--primary-teal)';

            event.target.style.color = 'white';

            

            setTimeout(() => {

                switch(path) {

                    case 'support':

                        alert('Routing to donation experience...');

                        showPage('donate');

                        break;

                    case 'church':

                        alert('Routing to US Church Partner experience...');

                        showPage('about');

                        break;

                    case 'business':

                        alert('Routing to business assessment...');

                        showPage('about');

                        break;

                    case 'explore':

                        alert('Routing to impact member experience...');

                        showPage('impact');

                        break;

                }

            }, 500);

        }
